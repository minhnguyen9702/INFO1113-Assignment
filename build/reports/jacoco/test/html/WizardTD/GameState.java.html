package WizardTD;
/**
 * Enum representing game's possible states.
 * 
 * - GAMEOVER: the player has lost.
 * - GAMEWIN: the player has won.
 * - GAMENORMAL: the player is still playing.
 */
public enum GameState {
    GAMEOVER, GAMEWIN, GAMENORMAL;
}
